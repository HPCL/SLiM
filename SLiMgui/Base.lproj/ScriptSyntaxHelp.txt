#SEX
<A|X|Y> [<x-dominance>]
// [<x-dominance>] is valid for X only, default 1.0
Example: A
// model an autosome in a sexual population
// (to model hermaphrodites, omit the #SEX tag)

#MUTATION TYPES
<mut-type> <h> <DFE-type: f|e|g> [DFE params]
// f <s>: fixed DFE, selection coefficient s
// e <mean-s>: exponential DFE with mean μ
// g <mean-s> <shape-alpha>: gamma DFE with μ and α
Example: m1 0.5 f 0.0
// neutral mutations: fixed DFE, 0.0 coefficient

#MUTATION RATE
<u>
Example: 1e-7
// set a rate of 1e-7 per nucleotide per generation

#GENOMIC ELEMENT TYPES
<element-type> <mut-type> <x> [<mut-type> <x> ...]
Example: g1 m1 1.0
// genomic element (g1) undergoing neutral mutations

#CHROMOSOME ORGANIZATION
<element-type> <start> <end>
Example: g1 1 100000
// uniform chromosome, length 100 kb, all of type g1

#RECOMBINATION RATE
<interval-end> <r>
Example: 100000 1e-8
// 1e-8 = 1 cM/Mb along the whole chromosome

#GENE CONVERSION
<fraction> <stretch-length>
Example: 0.1 1000
// 10% chance of conversion, 1000 is the mean length

#GENERATIONS
<t> [<start>]
Example: 10000
// run for 10000 generations; start at 1 by default

#DEMOGRAPHY AND STRUCTURE
<time> <event-type: P|N|M|S|X> [event parameters]
// P <pop> <N> [<source-pop>] [<X>]: new pop. of
//   <size>, split from <source-pop>, sex ratio <X>
// N <pop> <N>: change population size to <N>
// M <target-pop> <source-pop> <m>: set migration
//   rate <m> from <source-pop> into <target-pop>
// S <pop> <σ>: set selfing rate <σ> for <pop>
// X <pop> <X>: set sex ratio <X> for <pop>
Example: 1 P p1 500
// at generation 1, new population (p1), size 500

#OUTPUT
<time> <output-type: A|R|F|T> [output parameters]
// A [<filename>]: output all individuals [to file]
// R <pop> <size> [M|F|*] [MS]: output a sample of
//   <size> of the given gender, optional MS format
// F: output a list of all fixed mutations
// T <mut-type>: track and output a mutation-type
Example: 1000 R p1 10
// at gen. 1000, output a 10-genome sample from p1

#PREDETERMINED MUTATIONS
<time> <mut-type> <x> <pop> <nAA> <nAa> [P <f>]
Example: 100 m1 5000 0.01 0.5 p1 0 1
// ok, this is complicated; see the manual

#INITIALIZATION
<filename>   // initialize from the given file

#SEED
<seed>   // use a RNG seed for reproducibility
